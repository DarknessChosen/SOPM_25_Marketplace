<script>
function recalculate() {
  let table = document.getElementById('invoiceTable').getElementsByTagName('tbody')[0];
  let subtotal = 0;
  let totalTVA = 0;

  for (let row of table.rows) {
    let qty = parseFloat(row.cells[1].children[0].value) || 0;
    let price = parseFloat(row.cells[2].children[0].value) || 0;
    let tax = parseFloat(row.cells[3].children[0].value) || 0;

    let lineSubtotal = qty * price;
    let lineTVA = lineSubtotal * (tax / 100);
    let lineTotal = lineSubtotal + lineTVA;

    subtotal += lineSubtotal;
    totalTVA += lineTVA;

    row.cells[4].children[0].value = lineTotal.toFixed(2);
  }

  document.getElementById('subtotal').innerText = subtotal.toFixed(2);
  document.getElementById('tva').innerText = totalTVA.toFixed(2);
  document.getElementById('total').innerText = (subtotal + totalTVA).toFixed(2);
}

function deleteRow(btn) {
  let row = btn.parentNode.parentNode;
  row.parentNode.removeChild(row);
  recalculate();
}

window.onload = function() {
  const params = new URLSearchParams(window.location.search);
  const product = params.get("product");
  const price = parseFloat(params.get("price"));

  let tbody = document.querySelector("#invoiceTable tbody");

  if (product && price) {
    tbody.innerHTML = `
      <tr>
        <td><input type="text" value="${product}" readonly></td>
        <td><input type="number" value="1" min="1" onchange="recalculate()"></td>
        <td><input type="number" value="${price}" readonly></td>
        <td><input type="number" value="19" readonly></td>
        <td class="right"><input type="number" value="0.00" readonly></td>
        <td><button class="btn-delete" onclick="deleteRow(this)">»òterge</button></td>
      </tr>
    `;
  }

  recalculate();
};
</script>
